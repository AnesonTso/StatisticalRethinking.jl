    Define a structure to hold the data.

    struct BernoulliProblem
    "Total number of draws in the data."
    n::Int
    "Number of draws `==1` in the data"
    s::Vector{Int}
end;

    function (problem::BernoulliProblem)((Œ±, )::NamedTuple{(:Œ±, )})

end
