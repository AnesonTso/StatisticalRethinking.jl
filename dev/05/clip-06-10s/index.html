    Define the Stan language model

    m5_3 = "
data {
  int N;
  vector[N] divorce;
  vector[N] marriage_z;
  vector[N] median_age_z;
}
parameters {
  real a;
  real bA;
  real bM;
  real<lower=0> sigma;
}
model {
  vector[N] mu = a + median_age_z * bA + marriage_z * bM;
  sigma ~ uniform( 0 , 10 );
  bA ~ normal( 0 , 1 );
  bM ~ normal( 0 , 1 );
  a ~ normal( 10 , 10 );
  divorce ~ normal( mu , sigma );
}
";
    "\ndata {\n  int N;\n  vector[N] divorce;\n  vector[N] marriage_z;\n  vector[N] median_age_z;\n}\nparameters {\n  real a;\n  real bA;\n  real bM;\n  real<lower=0> sigma;\n}\nmodel {\n  vector[N] mu = a + median_age_z * bA + marriage_z * bM;\n  sigma ~ uniform( 0 , 10 );\n  bA ~ normal( 0 , 1 );\n  bM ~ normal( 0 , 1 );\n  a ~ normal( 10 , 10 );\n  divorce ~ normal( mu , sigma );\n}\n"

    Define the Stanmodel and set the output format to :mcmcchains.

    stanmodel = Stanmodel(name="m5_3",
monitors = ["a", "bA", "bM", "sigma", "Divorce"],
 model=m5_3, output_format=:mcmcchains);
